SELECT B.ESTORNO_FINANCEIRO,* 
  FROM VENDAS_ANALITICAS   A WITH(NOLOCK)
  JOIN DEV_PRODUTOS_CAIXAS B WITH(NOLOCK)ON A.FORMULARIO_ORIGEM = B.FORMULARIO_ORIGEM
                                        AND A.TAB_MASTER_ORIGEM = B.TAB_MASTER_ORIGEM
										AND A.REG_MASTER_ORIGEM = B.DEVOLUCAO_PRODUTO
 WHERE A.EMPRESA = 5 
   AND A.MOVIMENTO BETWEEN '01/05/2022' AND '31/05/2022'  
   AND A.VENDA_LIQUIDA < 0 
   AND B.ESTORNO_FINANCEIRO = 'N'
   AND NOT EXISTS (SELECT AA.DEVOLUCAO_PRODUTO FROM CANCELAMENTOS_VALE_CREDITO AA WITH(NOLOCK) WHERE AA.DEVOLUCAO_PRODUTO = B.DEVOLUCAO_PRODUTO)
GO
SET STATISTICS TIME OFF
