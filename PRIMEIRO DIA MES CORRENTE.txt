--ULTIMO DIA MES PASSADO
DECLARE @DATA DATETIME
SET @DATA = GETDATE()
SELECT CONVERT(VARCHAR,@DATA-DAY(GETDATE()),103) AS PRIMEIRO_DIA
 
GO
 
--PRIMEIRO DIA DE QUALQUER MES 
DECLARE @DATA DATETIME
SET @DATA='25-12-2019'
SELECT CONVERT(VARCHAR,@DATA-DATEPART (DAY,@DATA)+1,103) AS PRIMEIRO_DIA

GO

--PRIMEIRO DIA E ULTIMO DIA DO MES PASSADO
SELECT DATEADD(MONTH, DATEDIFF(MONTH,0,GETDATE()) - 1, 0) , 
       DATEADD(MONTH, DATEDIFF(MONTH,0,GETDATE()), -1)    ,
       EOMONTH(DATEADD(MONTH,1,GETDATE()))
